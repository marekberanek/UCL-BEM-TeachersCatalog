Draggable SVG images

var paper = Raphael( "tabs-raphael-2", 800, 200 );

var svgImages = [
  'images/daily-calendar2.svg',
  'images/done2.svg',
  'images/magnifyingglass.svg',
  'images/mail3.svg',
  'images/operations.svg'
];

var start = function() {
  this.ox = this.attr("x");
  this.oy = this.attr("y");
  this.animate({width: 48, height: 48}, 100, ">");
};
var move = function(dx, dy) {
  this.attr( {x: this.ox + dx, y: this.oy + dy} );
};
var up = function() {
  this.animate( {width: 32, height: 32}, 100, ">");
};

svgImages.forEach( function( svgImage, index ) {
  var imgEl = paper.image( svgImage, 40 + index * 64, 40, 32, 32 );
  imgEl.attr( 'cursor', 'move' );
  imgEl.drag( move, start, up );
});
